import numpy as np
import matplotlib.pyplot as plt
from skmultiflow.data import ConceptDriftStream
from skmultiflow.data import AGRAWALGenerator
from skmultiflow.meta import AdaptiveRandomForestClassifier
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score

# Function to run experiments
def run_experiment(drift_widths, drift_magnitudes, n_samples=1000):
    results = []

    # Loop over each combination of drift width and magnitude
    for width in drift_widths:
        for alpha in drift_magnitudes:
            # Create base stream and drift stream
            base_stream = AGRAWALGenerator(random_state=42)
            drift_stream = AGRAWALGenerator(random_state=99, classification_function=2)
            
            # Create the stream with a drift at position 500
            stream = ConceptDriftStream(stream=base_stream, drift_stream=drift_stream,
                                        position=500, width=width, alpha=alpha)
            
            # Simulate data from the stream
            X, y = [], []
            for i in range(n_samples):
                x_sample, y_sample = stream.next_sample()
                X.append(x_sample[0])
                y.append(y_sample[0])

            X = np.array(X)
            y = np.array(y)

            # Splitting into train and test sets (50% each)
            X_train, y_train = X[:500], y[:500]
            X_test, y_test = X[500:], y[500:]

            # Initialize classifiers
            random_forest = RandomForestClassifier(random_state=42)
            adaptive_rf = AdaptiveRandomForestClassifier(random_state=42, n_estimators=10)

            # Fit the models
            random_forest.fit(X_train, y_train)
            adaptive_rf.fit(X_train, y_train)

            # Test the models
            y_pred_rf = random_forest.predict(X_test)
            y_pred_adaptive_rf = adaptive_rf.predict(X_test)

            # Calculate accuracy
            accuracy_rf = accuracy_score(y_test, y_pred_rf)
            accuracy_adaptive_rf = accuracy_score(y_test, y_pred_adaptive_rf)

            # Store results
            results.append({
                'drift_width': width,
                'drift_magnitude': alpha,
                'random_forest_accuracy': accuracy_rf,
                'adaptive_random_forest_accuracy': accuracy_adaptive_rf
            })
            
            print(f"Drift Width: {width}, Drift Magnitude: {alpha}, RF Accuracy: {accuracy_rf:.2f}, ARF Accuracy: {accuracy_adaptive_rf:.2f}")
    
    return results

# Define drift widths and magnitudes
drift_widths = [0.00001, 100, 500, 1000, 2000, 3000]
drift_magnitudes = [0.00001, 0.5, 1.0]  # Different alpha values controlling drift magnitude

# Run the experiment
results = run_experiment(drift_widths, drift_magnitudes)

# Convert the results into a DataFrame for easier visualization
import pandas as pd
results_df = pd.DataFrame(results)

# Plot the results
plt.figure(figsize=(12, 8))
for alpha in drift_magnitudes:
    subset = results_df[results_df['drift_magnitude'] == alpha]
    plt.plot(subset['drift_width'], subset['random_forest_accuracy'], label=f'RF, Magnitude={alpha}', marker='o')
    plt.plot(subset['drift_width'], subset['adaptive_random_forest_accuracy'], label=f'ARF, Magnitude={alpha}', marker='x')

plt.title('Accuracy vs Drift Width for Different Drift Magnitudes')
plt.xlabel('Drift Width')
plt.ylabel('Accuracy')
plt.legend()
plt.grid(True)
plt.show()

# Display the results
print(results_df)
