import numpy as np
import matplotlib.pyplot as plt
from skmultiflow.data import ConceptDriftStream, SEAGenerator
from skmultiflow.drift_detection import DDM, ADWIN, EDDM
from skmultiflow.trees import HoeffdingTreeClassifier
from scipy.stats import ks_2samp

# Parameters
random_state = 42
concept_samples = 50000
drift_position = 25000  # Position where drift occurs
width = 1  # Width of the drift

# Create SEAGenerator with drift
generator_with_drift = ConceptDriftStream(
    stream=SEAGenerator(classification_function=0, random_state=random_state),
    drift_stream=SEAGenerator(classification_function=1, random_state=random_state),
    position=drift_position,
    width=width
)

# Create SEAGenerator with no drift
generator_no_drift = SEAGenerator(classification_function=0, random_state=random_state)

# Initialize Drift Detectors
ddm_with_drift = DDM()
adwin_with_drift = ADWIN()
eddm_with_drift = EDDM()
ddm_no_drift = DDM()
adwin_no_drift = ADWIN()
eddm_no_drift = EDDM()

# Initialize classifiers
clf_with_drift = HoeffdingTreeClassifier()
clf_no_drift = HoeffdingTreeClassifier()

# Initialize the classifiers with the first sample to avoid the error
X_initial, y_initial = generator_with_drift.next_sample()
clf_with_drift.partial_fit(X_initial, y_initial, classes=[0, 1])

X_initial_no_drift, y_initial_no_drift = generator_no_drift.next_sample()
clf_no_drift.partial_fit(X_initial_no_drift, y_initial_no_drift, classes=[0, 1])

# Variables to track drift detection and store labels for plotting
drift_points_with_drift_ddm = []
warning_points_with_drift_ddm = []
drift_points_no_drift_ddm = []
warning_points_no_drift_ddm = []

drift_points_with_drift_adwin = []
warning_points_with_drift_adwin = []
drift_points_no_drift_adwin = []
warning_points_no_drift_adwin = []

drift_points_with_drift_eddm = []
warning_points_with_drift_eddm = []
drift_points_no_drift_eddm = []
warning_points_no_drift_eddm = []

# Lists to store labels for plotting
y_labels_with_drift = [y_initial[0]]
y_labels_no_drift = [y_initial_no_drift[0]]

buffer_with_drift = []
buffer_no_drift = []

# Process the data stream and apply DDM, ADWIN, EDDM, and KS (with drift)
for i in range(1, concept_samples):  # Start from 1 because we've used the first sample already
    X, y = generator_with_drift.next_sample()
    y_pred = clf_with_drift.predict(X)[0]
    error = int(y_pred != y[0])
    
    # Update detectors with error
    ddm_with_drift.add_element(error)
    adwin_with_drift.add_element(error)
    eddm_with_drift.add_element(error)
    clf_with_drift.partial_fit(X, y)
    
    # Append the label for plotting
    y_labels_with_drift.append(y[0])
    
    # DDM Drift Detection
    if ddm_with_drift.detected_warning_zone():
        warning_points_with_drift_ddm.append(i)
    if ddm_with_drift.detected_change():
        drift_points_with_drift_ddm.append(i)
    
    # ADWIN Drift and Warning Detection
    if adwin_with_drift.detected_warning_zone():
        warning_points_with_drift_adwin.append(i)
    if adwin_with_drift.detected_change():
        drift_points_with_drift_adwin.append(i)
    
    # EDDM Drift Detection
    if eddm_with_drift.detected_warning_zone():
        warning_points_with_drift_eddm.append(i)
    if eddm_with_drift.detected_change():
        drift_points_with_drift_eddm.append(i)

# Process the data stream and apply DDM, ADWIN, EDDM, and KS (no drift)
for i in range(1, concept_samples):  # Start from 1 because we've used the first sample already
    X, y = generator_no_drift.next_sample()
    y_pred = clf_no_drift.predict(X)[0]
    error = int(y_pred != y[0])
    
    # Update detectors with error
    ddm_no_drift.add_element(error)
    adwin_no_drift.add_element(error)
    eddm_no_drift.add_element(error)
    clf_no_drift.partial_fit(X, y)
    
    # Append the label for plotting
    y_labels_no_drift.append(y[0])
    
    # DDM Drift Detection
    if ddm_no_drift.detected_warning_zone():
        warning_points_no_drift_ddm.append(i)
    if ddm_no_drift.detected_change():
        drift_points_no_drift_ddm.append(i)
    
    # ADWIN Drift and Warning Detection
    if adwin_no_drift.detected_warning_zone():
        warning_points_no_drift_adwin.append(i)
    if adwin_no_drift.detected_change():
        drift_points_no_drift_adwin.append(i)
    
    # EDDM Drift Detection
    if eddm_no_drift.detected_warning_zone():
        warning_points_no_drift_eddm.append(i)
    if eddm_no_drift.detected_change():
        drift_points_no_drift_eddm.append(i)

plt.figure(figsize=(12, 6))
plt.scatter(range(concept_samples), y_labels_with_drift, c='blue', label='Data with Drift', s=2)
